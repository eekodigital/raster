---
title: ScrollArea
description: A scrollable container with custom-styled scrollbars that overlay content. Built on Radix ScrollArea for cross-browser consistency and accessibility. The scrollbar appears on hover and auto-hides when not in use.
---

import Example from "../../../components/Example.astro";
import {
  ScrollAreaBasicDemo,
  ScrollAreaHorizontalDemo,
} from "../../../components/ScrollAreaDemo.tsx";

## Vertical scroll

<Example
  code={`import { ScrollArea } from '@eekodigital/raster';\n\n<ScrollArea.Root style={{ height: '12rem', width: '22rem' }}>\n  <ScrollArea.Viewport>\n    {items.map(item => <div key={item}>{item}</div>)}\n  </ScrollArea.Viewport>\n  <ScrollArea.Scrollbar orientation="vertical" />\n  <ScrollArea.Corner />\n</ScrollArea.Root>`}
>
  <ScrollAreaBasicDemo client:load />
</Example>

## Horizontal scroll

<Example
  code={`<ScrollArea.Root style={{ width: '22rem' }}>\n  <ScrollArea.Viewport>\n    <div style={{ display: 'flex', gap: '0.75rem', width: 'max-content' }}>\n      {items.map(item => <div key={item}>{item}</div>)}\n    </div>\n  </ScrollArea.Viewport>\n  <ScrollArea.Scrollbar orientation="horizontal" />\n</ScrollArea.Root>`}
