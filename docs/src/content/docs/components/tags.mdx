---
title: Tag
description: A small inline label used to communicate status, category, or metadata at a glance. Tags are purely presentational — they do not trigger actions.
---

import Example from "../../../components/Example.astro";
import { Tag } from "@eekodigital/raster";

## Variants

Choose the variant that best communicates the semantic intent of the label.

### Neutral

<Example code={`import { Tag } from '@eekodigital/raster';\n\n<Tag>Archived</Tag>`}>
  <Tag client:load>Archived</Tag>
</Example>

### Success

<Example code={`<Tag variant="success">Published</Tag>`}>
  <Tag client:load variant="success">
    Published
  </Tag>
</Example>

### Danger

<Example code={`<Tag variant="danger">Overdue</Tag>`}>
  <Tag client:load variant="danger">
    Overdue
  </Tag>
</Example>

### Warning

<Example code={`<Tag variant="warning">Pending review</Tag>`}>
  <Tag client:load variant="warning">
    Pending review
  </Tag>
</Example>

### Info

<Example code={`<Tag variant="info">In progress</Tag>`}>
  <Tag client:load variant="info">
    In progress
  </Tag>
</Example>
