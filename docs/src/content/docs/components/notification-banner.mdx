---
title: NotificationBanner
description: A persistent, inline message used to communicate important status information to the user. Unlike Toast, a banner remains visible until the underlying condition is resolved.
---

import { NotificationBanner as NotificationBannerDS } from "@eekodigital/raster";
import Example from "../../../components/Example.astro";

<div class="callout">
  <p>
    <strong>Banner vs Toast.</strong> Use <code>NotificationBanner</code> for persistent, page-level
    messages that describe the current state (e.g. a form submission error or a maintenance
    warning). Use <code>Toast</code> for ephemeral confirmations of a completed action (e.g. "Report
    saved").
  </p>
</div>

## Info

<Example
  code={`import { NotificationBanner } from '@eekodigital/raster';\n\n<NotificationBanner variant="info" title="Scheduled maintenance">\n  The system will be unavailable on Saturday 29 March between 02:00 and 04:00.\n</NotificationBanner>`}
>
  <NotificationBannerDS client:load variant="info" title="Scheduled maintenance">
    The system will be unavailable on Saturday 29 March between 02:00 and 04:00.
  </NotificationBannerDS>
</Example>

## Title only

Omit `children` when a short heading is sufficient.

<Example code={`<NotificationBanner variant="success" title="Changes saved successfully" />`}>
  <NotificationBannerDS client:load variant="success" title="Changes saved successfully" />
</Example>

## Body only

Omit `title` for a plain message without a heading.

<Example
  code={`<NotificationBanner variant="info" aria-label="Session expiry notice">\n  Your session will expire in 5 minutes.\n</NotificationBanner>`}
>
  <NotificationBannerDS client:load variant="info" aria-label="Session expiry notice">
    Your session will expire in 5 minutes.
  </NotificationBannerDS>
</Example>
