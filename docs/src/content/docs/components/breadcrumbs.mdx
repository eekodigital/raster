---
title: Breadcrumbs
description: A navigational trail showing the user's position within a site hierarchy. The last item represents the current page and is rendered as plain text rather than a link.
---

import Example from "../../../components/Example.astro";
import { Breadcrumbs as BreadcrumbsDS } from "@eekodigital/raster";

## Basic trail

Pass an `items` array of objects. The final item without an `href` is treated as the current page.

<Example
  code={`import { Breadcrumbs } from '@eekodigital/raster';\n\n<Breadcrumbs\n  items={[\n    { label: 'Home', href: '/' },\n    { label: 'Reports', href: '/reports' },\n    { label: 'Annual report 2024' },\n  ]}\n/>`}
>
  <BreadcrumbsDS
    client:load
    aria-label="Breadcrumb: basic trail"
    items={[
      { label: "Home", href: "/" },
      { label: "Reports", href: "/reports" },
      { label: "Annual report 2024" },
    ]}
  />
</Example>

## Single item

When the user is at the top level, a single item is sufficient.

<Example code={`<Breadcrumbs items={[{ label: 'Dashboard' }]} />`}>
  <BreadcrumbsDS
    client:load
    aria-label="Breadcrumb: single item"
    items={[{ label: "Dashboard" }]}
  />
</Example>

## Longer trail

<Example
  code={`<Breadcrumbs\n  items={[\n    { label: 'Home', href: '/' },\n    { label: 'Compliance', href: '/compliance' },\n    { label: 'Audits', href: '/compliance/audits' },\n    { label: '2024', href: '/compliance/audits/2024' },\n    { label: 'Q1 review' },\n  ]}\n/>`}
>
  <BreadcrumbsDS
    client:load
    aria-label="Breadcrumb: longer trail"
    items={[
      { label: "Home", href: "/" },
      { label: "Compliance", href: "/compliance" },
      { label: "Audits", href: "/compliance/audits" },
      { label: "2024", href: "/compliance/audits/2024" },
      { label: "Q1 review" },
    ]}
  />
</Example>
