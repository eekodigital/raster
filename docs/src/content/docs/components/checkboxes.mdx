import Example from "../../../components/Example.astro";
import { Checkbox } from "@eekodigital/raster";
import { CheckboxGroupDemo, CheckboxGroupHintErrorDemo } from "../../../components/CheckboxDemo";

## Single checkbox

<Example
  code={`import { Checkbox } from '@eekodigital/raster';\n\n<Checkbox label="Send me a summary email" name="email_summary" />`}
>
  <Checkbox client:load label="Send me a summary email" name="email_summary" />
</Example>

## CheckboxGroup

<Example
  code={`<CheckboxGroup legend="Notify me about">\n  <Checkbox label="New reports" name="notify_reports" />\n  <Checkbox label="Criterion status changes" name="notify_status" />\n  <Checkbox label="Report locks and unlocks" name="notify_locks" />\n</CheckboxGroup>`}
>
  <CheckboxGroupDemo client:load />
</Example>

## With hint and error

<Example
  code={`<CheckboxGroup\n  legend="Notify me about"\n  hint="You can change this later in settings."\n  error="Select at least one notification type."\n>\n  <Checkbox label="New reports" name="notify_reports" />\n  <Checkbox label="Criterion status changes" name="notify_status" />\n</CheckboxGroup>`}
>
  <CheckboxGroupHintErrorDemo client:load />
</Example>
