---
title: Select
description: Two levels of the same control. Select is the bare styled select — use it for custom layouts, table cells, or inline editors. SelectField adds a label, optional hint, and optional error for the standard form field pattern.
---

  SelectDemo,
  SelectErrorDemo,

} from "../../../components/SelectDemo.tsx";

## SelectField

### Default

<Example code={`import { SelectField } from '@eekodigital/raster';

<SelectField
  label="Standard"
  name="standard"
  placeholder="Choose a standard…"
  options={[
    { value: 'wcag-2.2', label: 'WCAG 2.2' },
    { value: 'wcag-3.0', label: 'WCAG 3.0 (beta)' },
  ]}
/>`}>
  <SelectField
    client:only="react"
    label="Standard"
    name="standard"
    placeholder="Choose a standard…"
    options={[
      { value: 'wcag-2.2', label: 'WCAG 2.2' },
      { value: 'wcag-3.0', label: 'WCAG 3.0 (beta)' },
    ]}
  />
</Example>

### With hint

<Example
  code={`<SelectField label="Standard" name="standard" hint="The standard all results will be evaluated against." options={[...]} />`}
>
  <SelectField
    client:only="react"
    label="Standard"
    name="standard"
    hint="The standard all results in this audit will be evaluated against."
    placeholder="Choose a standard…"
    options={[
      { value: "wcag-2.2", label: "WCAG 2.2" },
      { value: "wcag-3.0", label: "WCAG 3.0 (beta)" },
    ]}
  />
</Example>
